/app/frontend/src/pages/Dashboard.js                                                                               import React, { useEffect, useState } from 'react';
import { motion } from 'framer-motion';
import { useAuth } from '../contexts/AuthContext';
import axios from 'axios';
import { toast } from 'sonner';
import { CheckCircle, Clock, FileText, Award } from 'lucide-react';
import { useNavigate } from 'react-router-dom';

const BACKEND_URL = process.env.REACT_APP_BACKEND_URL;
const API = `${BACKEND_URL}/api`;

const statusConfig = {
  pending: { label: 'Pending', step: 1, icon: Clock, color: 'yellow' },
  reviewed: { label: 'Under Review', step: 2, icon: FileText, color: 'blue' },
  shortlisted: { label: 'Shortlisted', step: 3, icon: Award, color: 'purple' },
  accepted: { label: 'Accepted', step: 4, icon: CheckCircle, color: 'green' },
  rejected: { label: 'Not Selected', step: 4, icon: Clock, color: 'red' }
};

const ProgressStepper = ({ status }) => {
  const config = statusConfig[status] || statusConfig.pending;
  const steps = ['Submitted', 'Review', 'Shortlist', 'Decision'];
  
  return (
    <div className="w-full py-4">
      <div className="flex items-center justify-between">
        {steps.map((step, index) => {
          const stepNumber = index + 1;
          const isCompleted = stepNumber < config.step || status === 'accepted';
          const isCurrent = stepNumber === config.step;
          const isRejected = status === 'rejected' && stepNumber === config.step;
          
          return (
            <React.Fragment key={step}>
              <div className="flex flex-col items-center">
                <div
                  className={`w-10 h-10 rounded-full flex items-center justify-center font-semibold transition-all ${
                    isCompleted
                      ? 'bg-green-700 text-white'
                      : isCurrent && !isRejected
                      ? 'bg-slate-900 text-white'
                      : isRejected
                      ? 'bg-red-700 text-white'
                      : 'bg-slate-200 text-slate-400'
                  }`}
                >
                  {isCompleted ? '✓' : stepNumber}
                </div>
                <span className={`text-xs mt-2 font-sans ${
                  isCompleted || isCurrent ? 'text-slate-900 font-medium' : 'text-slate-400'
                }`}>
                  {step}
                </span>
              </div>
              {index < steps.length - 1 && (
                <div className={`flex-1 h-0.5 mx-2 transition-all ${
                  isCompleted ? 'bg-green-700' : 'bg-slate-200'
                }`} />
              )}
            </React.Fragment>
          );
        })}
      </div>
    </div>
  );
};

const Dashboard = () => {
  const { getAuthHeader, user } = useAuth();
  const [applications, setApplications] = useState([]);
  const [loading, setLoading] = useState(true);
  const navigate = useNavigate();

  useEffect(() => {
    fetchApplications();
  }, []);

  const fetchApplications = async () => {
    try {
      const response = await axios.get(`${API}/my-applications`, {
        headers: getAuthHeader()
      });
      setApplications(response.data);
    } catch (error) {
      toast.error('Failed to load applications');
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-stone-50 flex items-center justify-center">
        <div className="text-slate-600 font-sans">Loading your applications...</div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-stone-50 p-6 md:p-12">
      <div className="max-w-7xl mx-auto">
        <div className="mb-8">
          <h1 className="font-serif text-4xl sm:text-5xl font-bold text-slate-900 mb-2">
            My Applications
          </h1>
          <p className="text-slate-600 font-sans text-lg">Track your scholarship and internship applications</p>
        </div>

        {applications.length === 0 ? (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            className="bg-white border border-slate-200 rounded-xl p-12 text-center"
          >
            <FileText className="w-16 h-16 text-slate-300 mx-auto mb-4" />
            <h3 className="font-serif text-xl font-bold text-slate-900 mb-2">No Applications Yet</h3>
            <p className="text-slate-600 font-sans mb-6">Start your journey by browsing available opportunities</p>
            <button
              data-testid="browse-opportunities-button"
              onClick={() => navigate('/opportunities')}
              className="rounded-full bg-slate-900 text-white hover:bg-slate-800 transition-all shadow-lg hover:shadow-xl px-8 py-3 font-medium"
            >
              Browse Opportunities
            </button>
          </motion.div>
        ) : (
          <div className="grid grid-cols-1 gap-6">
            {applications.map((app, index) => {
              const config = statusConfig[app.status] || statusConfig.pending;
              const Icon = config.icon;
              
              return (
                <motion.div
                  key={app.id}
                  initial={{ opacity: 0, y: 20 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ delay: index * 0.1 }}
                  className="bg-white border border-slate-200 rounded-xl p-6 hover:border-slate-300 transition-colors shadow-sm"
                  data-testid={`application-card-${index}`}
                >
                  <div className="flex flex-col md:flex-row md:items-start md:justify-between mb-6">
                    <div className="flex-1">
                      <h3 className="font-serif text-xl font-bold text-slate-900 mb-1">
                        {app.posting?.title || 'Unknown Posting'}
                      </h3>
                      <p className="text-slate-600 font-sans text-sm">
                        {app.posting?.organization} • {app.posting?.type === 'scholarship' ? 'Scholarship' : 'Internship'}
                      </p>
                    </div>
                    <div className="mt-3 md:mt-0">
                      <span
                        className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium bg-${config.color}-100 text-${config.color}-800`}
                        data-testid={`application-status-${index}`}
                      >
                        <Icon className="w-3 h-3 mr-1" />
                        {config.label}
                      </span>
                    </div>
                  </div>

                  <ProgressStepper status={app.status} />

                  <div className="mt-6 pt-6 border-t border-slate-100 grid grid-cols-2 gap-4 text-sm">
                    <div>
                      <span className="text-xs uppercase tracking-widest font-semibold text-slate-500">Submitted</span>
                      <p className="font-sans text-slate-900 mt-1">
                        {new Date(app.submitted_at).toLocaleDateString('en-US', {
                          month: 'short',
                          day: 'numeric',
                          year: 'numeric'
                        })}
                      </p>
                    </div>
                    <div>
                      <span className="text-xs uppercase tracking-widest font-semibold text-slate-500">Deadline</span>
                      <p className="font-sans text-slate-900 mt-1">
                        {app.posting?.deadline
                          ? new Date(app.posting.deadline).toLocaleDateString('en-US', {
                              month: 'short',
                              day: 'numeric',
                              year: 'numeric'
                            })
                          : 'N/A'}
                      </p>
                    </div>
                  </div>
                </motion.div>
              );
            })}
          </div>
        )}
      </div>
    </div>
  );
};

export default Dashboard;
